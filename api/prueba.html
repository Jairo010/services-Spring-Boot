function editUser() { var row = $('#dg').datagrid('getSelected'); $.ajax({ url: 'http://quintosoa.com/api.php?cedula=' + row.cedula, type: 'DELETE', success: function(result) { if (result.success) { $.messager.show({ title: 'Error', msg: result.errorMsg }); } else { $('#dg').datagrid('reload'); } }, dataType: 'json' }); if (row) { $('#dlg').dialog('open').dialog('center').dialog('setTitle', 'Edit User'); $('#fm').form('load', row); url = 'http://quintosoa.com/api.php'; } }
